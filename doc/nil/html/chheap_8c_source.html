149 /**
150  * @brief Allocates a block of memory from the heap by using the first-fit
151  * algorithm.
152  * @details The allocated block is guaranteed to be properly aligned to the
153  * specified alignment.
154  *
155  * @param[in] heapp pointer to a heap descriptor or @p NULL in order to
156  * access the default heap.
157  * @param[in] size the size of the block to be allocated. Note that the
158  * allocated block may be a bit bigger than the requested
159  * size for alignment and fragmentation reasons.
160  * @param[in] align desired memory alignment
161  * @return A pointer to the aligned allocated block.
162  * @retval NULL if the block cannot be allocated.
163  *
164  * @api
165  */
166 void *chHeapAllocAligned(memory_heap_t *heapp, size_t size, unsigned align) {
167  heap_header_t *qp, *hp;
168  size_t pages;
169 
170  chDbgCheck((size > 0U) && MEM_IS_VALID_ALIGNMENT(align));
171 
172  /* If an heap is not specified then the default system header is used.*/
173  if (heapp == NULL) {
174  heapp = &default_heap;
175  }
176 
177  /* Minimum alignment is constrained by the heap header structure size.*/
178  if (align < CH_HEAP_ALIGNMENT) {
179  align = CH_HEAP_ALIGNMENT;
180  }
181 
182  /* Size is converted in number of elementary allocation units.*/
183  pages = MEM_ALIGN_NEXT(size, CH_HEAP_ALIGNMENT) / CH_HEAP_ALIGNMENT;
184 
185  /* Taking heap mutex/semaphore.*/
186  H_LOCK(heapp);
187 
188  /* Start of the free blocks list.*/
189  qp = &heapp->header;
190  while (H_NEXT(qp) != NULL) {
191  heap_header_t *ahp;
192 
193  /* Next free block.*/
194  hp = H_NEXT(qp);
195 
196  /* Pointer aligned to the requested alignment.*/
197  ahp = (heap_header_t *)MEM_ALIGN_NEXT(H_BLOCK(hp), align) - 1U;
198 
199  if ((ahp < H_LIMIT(hp)) && (pages <= NPAGES(H_LIMIT(hp), ahp + 1U))) {
200  /* The block is large enough to contain a correctly aligned area
201  of sufficient size.*/
202 
203  if (ahp > hp) {
204  /* The block is not properly aligned, must split it.*/
205  size_t bpages;
206 
207  bpages = NPAGES(H_LIMIT(hp), H_BLOCK(ahp));
208  H_PAGES(hp) = NPAGES(ahp, H_BLOCK(hp));
209  if (bpages > pages) {
210  /* The block is bigger than required, must split the excess.*/
211  heap_header_t *fp;
212 
213  /* Creating the excess block.*/
214  fp = H_BLOCK(ahp) + pages;
215  H_PAGES(fp) = (bpages - pages) - 1U;
216 
217  /* Linking the excess block.*/
218  H_NEXT(fp) = H_NEXT(hp);
219  H_NEXT(hp) = fp;
220  }
221 
222  hp = ahp;
223  }
224  else {
225  /* The block is already properly aligned.*/
226 
227  if (H_PAGES(hp) == pages) {
228  /* Exact size, getting the whole block.*/
229  H_NEXT(qp) = H_NEXT(hp);
230  }
231  else {
232  /* The block is bigger than required, must split the excess.*/
233  heap_header_t *fp;
234 
235  fp = H_BLOCK(hp) + pages;
236  H_NEXT(fp) = H_NEXT(hp);
237  H_PAGES(fp) = NPAGES(H_LIMIT(hp), H_BLOCK(fp));
238  H_NEXT(qp) = fp;
239  }
240  }
241 
242  /* Setting in the block owner heap and size.*/
243  H_SIZE(hp) = size;
244  H_HEAP(hp) = heapp;
245 
246  /* Releasing heap mutex/semaphore.*/
247  H_UNLOCK(heapp);
248 
249  /*lint -save -e9087 [11.3] Safe cast.*/
250  return (void *)H_BLOCK(hp);
251  /*lint -restore*/
252  }
253 
254  /* Next in the free blocks list.*/
255  qp = hp;
256  }
257 
258  /* Releasing heap mutex/semaphore.*/
259  H_UNLOCK(heapp);
260 
261  /* More memory is required, tries to get it from the associated provider
262  else fails.*/
263  if (heapp->provider != NULL) {
264  hp = heapp->provider((pages + 1U) * CH_HEAP_ALIGNMENT, align);
265  if (hp != NULL) {
266  H_HEAP(hp) = heapp;
267  H_SIZE(hp) = size;
268 
269  /*lint -save -e9087 [11.3] Safe cast.*/
270  return (void *)H_BLOCK(hp);
271  /*lint -restore*/
272  }
273  }
274 
275  return NULL;
276 }

278 /**
279  * @brief Frees a previously allocated memory block.
280  *
281  * @param[in] p pointer to the memory block to be freed
282  *
283  * @api
284  */
285 void chHeapFree(void *p) {
286  heap_header_t *qp, *hp;
287  memory_heap_t *heapp;
288 
289  chDbgCheck((p != NULL) && MEM_IS_ALIGNED(p, CH_HEAP_ALIGNMENT));
290 
291  /*lint -save -e9087 [11.3] Safe cast.*/
292  hp = (heap_header_t *)p - 1U;
293  /*lint -restore*/
294  heapp = H_HEAP(hp);
295  qp = &heapp->header;
296 
297  /* Size is converted in number of elementary allocation units.*/
298  H_PAGES(hp) = MEM_ALIGN_NEXT(H_SIZE(hp),
299  CH_HEAP_ALIGNMENT) / CH_HEAP_ALIGNMENT;
300 
301  /* Taking heap mutex/semaphore.*/
302  H_LOCK(heapp);
303 
304  while (true) {
305  chDbgAssert((hp < qp) || (hp >= H_LIMIT(qp)), "within free block");
306 
307  if (((qp == &heapp->header) || (hp > qp)) &&
308  ((H_NEXT(qp) == NULL) || (hp < H_NEXT(qp)))) {
309  /* Insertion after qp.*/
310  H_NEXT(hp) = H_NEXT(qp);
311  H_NEXT(qp) = hp;
312  /* Verifies if the newly inserted block should be merged.*/
313  if (H_LIMIT(hp) == H_NEXT(hp)) {
314  /* Merge with the next block.*/
315  H_PAGES(hp) += H_PAGES(H_NEXT(hp)) + 1U;
316  H_NEXT(hp) = H_NEXT(H_NEXT(hp));
317  }
318  if ((H_LIMIT(qp) == hp)) {
319  /* Merge with the previous block.*/
320  H_PAGES(qp) += H_PAGES(hp) + 1U;
321  H_NEXT(qp) = H_NEXT(hp);
322  }
323  break;
324  }
325  qp = H_NEXT(qp);
326  }
327 
328  /* Releasing heap mutex/semaphore.*/
329  H_UNLOCK(heapp);
330 
331  return;
332 }

334 /**
335  * @brief Reports the heap status.
336  * @note This function is meant to be used in the test suite, it should
337  * not be really useful for the application code.
338  *
339  * @param[in] heapp pointer to a heap descriptor or @p NULL in order to
340  * access the default heap.
341  * @param[in] totalp pointer to a variable that will receive the total
342  * fragmented free space or @ NULL
343  * @param[in] largestp pointer to a variable that will receive the largest
344  * free free block found space or @ NULL
345  * @return The number of fragments in the heap.
346  *
347  * @api
348  */
349 size_t chHeapStatus(memory_heap_t *heapp, size_t *totalp, size_t *largestp) {
350  heap_header_t *qp;
351  size_t n, tpages, lpages;
352 
353  if (heapp == NULL) {
354  heapp = &default_heap;
355  }
356 
357  H_LOCK(heapp);
358  tpages = 0U;
359  lpages = 0U;
360  n = 0U;
361  qp = &heapp->header;
362  while (H_NEXT(qp) != NULL) {
363  size_t pages = H_PAGES(H_NEXT(qp));
364 
365  /* Updating counters.*/
366  n++;
367  tpages += pages;
368  if (pages > lpages) {
369  lpages = pages;
370  }
371 
372  qp = H_NEXT(qp);
373  }
374 
375  /* Writing out fragmented free memory.*/
376  if (totalp != NULL) {
377  *totalp = tpages * CH_HEAP_ALIGNMENT;
378  }
379 
380  /* Writing out unfragmented free memory.*/
381  if (largestp != NULL) {
382  *largestp = lpages * CH_HEAP_ALIGNMENT;
383  }
384  H_UNLOCK(heapp);
385 
386  return n;
387 }
