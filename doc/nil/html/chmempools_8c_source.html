20 /**
21  * @file chmempools.c
22  * @brief Memory Pools code.
23  *
24  * @addtogroup pools
25  * @details Memory Pools related APIs and services.
26  * <h2>Operation mode</h2>
27  * The Memory Pools APIs allow to allocate/free fixed size objects in
28  * <b>constant time</b> and reliably without memory fragmentation
29  * problems.<br>
30  * Memory Pools do not enforce any alignment constraint on the
31  * contained object however the objects must be properly aligned
32  * to contain a pointer to void.
33  * @pre In order to use the memory pools APIs the @p CH_CFG_USE_MEMPOOLS option
34  * must be enabled in @p chconf.h.
35  * @note Compatible with RT and NIL.
36  * @{
37  */

63 /**
64  * @brief Initializes an empty memory pool.
65  *
66  * @param[out] mp pointer to a @p memory_pool_t structure
67  * @param[in] size the size of the objects contained in this memory pool,
68  * the minimum accepted size is the size of a pointer to
69  * void.
70  * @param[in] provider memory provider function for the memory pool or
71  * @p NULL if the pool is not allowed to grow
72  * automatically
73  *
74  * @init
75  */
76 void chPoolObjectInit(memory_pool_t *mp, size_t size, memgetfunc_t provider) {
77 
78  chDbgCheck((mp != NULL) && (size >= sizeof(void *)));
79 
80  mp->next = NULL;
81  mp->object_size = size;
82  mp->provider = provider;
83 }

85 /**
86  * @brief Loads a memory pool with an array of static objects.
87  * @pre The memory pool must be already been initialized.
88  * @pre The array elements must be of the right size for the specified
89  * memory pool.
90  * @post The memory pool contains the elements of the input array.
91  *
92  * @param[in] mp pointer to a @p memory_pool_t structure
93  * @param[in] p pointer to the array first element
94  * @param[in] n number of elements in the array
95  *
96  * @api
97  */
98 void chPoolLoadArray(memory_pool_t *mp, void *p, size_t n) {
99 
100  chDbgCheck((mp != NULL) && (n != 0U));
101 
102  while (n != 0U) {
103  chPoolAdd(mp, p);
104  /*lint -save -e9087 [11.3] Safe cast.*/
105  p = (void *)(((uint8_t *)p) + mp->object_size);
106  /*lint -restore*/
107  n--;
108  }
109 }

111 /**
112  * @brief Allocates an object from a memory pool.
113  * @pre The memory pool must be already been initialized.
114  *
115  * @param[in] mp pointer to a @p memory_pool_t structure
116  * @return The pointer to the allocated object.
117  * @retval NULL if pool is empty.
118  *
119  * @iclass
120  */
121 void *chPoolAllocI(memory_pool_t *mp) {
122  void *objp;
123 
124  chDbgCheckClassI();
125  chDbgCheck(mp != NULL);
126 
127  objp = mp->next;
128  /*lint -save -e9013 [15.7] There is no else because it is not needed.*/
129  if (objp != NULL) {
130  mp->next = mp->next->next;
131  }
132  else if (mp->provider != NULL) {
133  objp = mp->provider(mp->object_size, PORT_NATURAL_ALIGN); /* TODO: Alignment is not properly handled */
134  }
135  /*lint -restore*/
136 
137  return objp;
138 }

160 /**
161  * @brief Releases an object into a memory pool.
162  * @pre The memory pool must be already been initialized.
163  * @pre The freed object must be of the right size for the specified
164  * memory pool.
165  * @pre The added object must be properly aligned.
166  *

171  */
172 void chPoolFreeI(memory_pool_t *mp, void *objp) {
173  struct pool_header *php = objp;
174 
175  chDbgCheckClassI();
176  chDbgCheck((mp != NULL) && (objp != NULL));
177 
178  php->next = mp->next;
179  mp->next = php;
180 }

244 /**
245  * @brief Allocates an object from a guarded memory pool.
246  * @pre The guarded memory pool must be already been initialized.
247  *
248  * @param[in] gmp pointer to a @p guarded_memory_pool_t structure
249  * @param[in] timeout the number of ticks before the operation timeouts,
250  * the following special values are allowed:
251  * - @a TIME_IMMEDIATE immediate timeout.
252  * - @a TIME_INFINITE no timeout.
253  * .
254  * @return The pointer to the allocated object.
255  * @retval NULL if the operation timed out.
256  *
257  * @sclass
258  */
259 void *chGuardedPoolAllocTimeoutS(guarded_memory_pool_t *gmp,
260  systime_t timeout) {
261  msg_t msg;
262 
263  msg = chSemWaitTimeoutS(&gmp->sem, timeout);
264  if (msg != MSG_OK) {
265  return NULL;
266  }
267 
268  return chPoolAllocI(&gmp->pool);
269 }
