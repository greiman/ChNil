213 /**
214  * @brief Initializes the kernel.
215  * @details Initializes the kernel structures, the current instructions flow
216  * becomes the idle thread upon return. The idle thread must not
217  * invoke any kernel primitive able to change state to not runnable.
218  * @note This function assumes that the @p nil global variable has been
219  * zeroed by the runtime environment. If this is not the case then
220  * make sure to clear it before calling this function.
221  *
222  * @special
223  */
224 void chSysInit(void) {
225  thread_t *tp;
226  const thread_config_t *tcp;
227 #if WHG_MOD
228  nil.threads = nil_threads;
229 #endif // WHG_MOD
230 
231 #if CH_DBG_SYSTEM_STATE_CHECK == TRUE
232  nil.isr_cnt = (cnt_t)0;
233  nil.lock_cnt = (cnt_t)0;
234 #endif
235 
236  /* System initialization hook.*/
237  CH_CFG_SYSTEM_INIT_HOOK();
238 
239  /* Iterates through the list of defined threads.*/
240  tp = &nil.threads[0];
241  tcp = nil_thd_configs;
242  while (tp < &nil.threads[CH_CFG_NUM_THREADS]) {
243 #if CH_DBG_ENABLE_STACK_CHECK
244  tp->wabase = (stkalign_t *)tcp->wbase;
245 #endif
246 
247  /* Port dependent thread initialization.*/
248  PORT_SETUP_CONTEXT(tp, tcp->wbase, tcp->wend, tcp->funcp, tcp->arg);
249 
250  /* Initialization hook.*/
251  CH_CFG_THREAD_EXT_INIT_HOOK(tp);
252 
253  tp++;
254  tcp++;
255  }
256 
257 #if CH_DBG_ENABLE_STACK_CHECK
258  /* The idle thread is a special case because its stack is set up by the
259  runtime environment.*/
260  tp->wabase = THD_IDLE_BASE;
261 #endif
262 
263  /* Interrupts partially enabled. It is equivalent to entering the
264  kernel critical zone.*/
265  chSysSuspend();
266 #if CH_DBG_SYSTEM_STATE_CHECK == TRUE
267  nil.lock_cnt = (cnt_t)1;
268 #endif
269 
270  /* Heap initialization, if enabled.*/
271 #if CH_CFG_USE_HEAP == TRUE
272  _heap_init();
273 #endif
274 
275  /* Port layer initialization last because it depend on some of the
276  initializations performed before.*/
277  port_init();
278 
279  /* Runs the highest priority thread, the current one becomes the idle
280  thread.*/
281  nil.current = nil.next = nil.threads;
282  port_switch(nil.current, tp);
283  chSysUnlock();
284 }

316 /**
317  * @brief Time management handler.
318  * @note This handler has to be invoked by a periodic ISR in order to
319  * reschedule the waiting threads.
320  *
321  * @iclass
322  */
323 void chSysTimerHandlerI(void) {
324 
325  chDbgCheckClassI();
326 
327 #if CH_CFG_ST_TIMEDELTA == 0
328  thread_t *tp = &nil.threads[0];
329  nil.systime++;
330  do {
331  /* Is the thread in a wait state with timeout?.*/
332  if (tp->timeout > (systime_t)0) {
333 
334  chDbgAssert(!NIL_THD_IS_READY(tp), "is ready");
335 
336  /* Did the timer reach zero?*/
337  if (--tp->timeout == (systime_t)0) {
338  /* Timeout on semaphores requires a special handling because the
339  semaphore counter must be incremented.*/
340  /*lint -save -e9013 [15.7] There is no else because it is not needed.*/
341  if (NIL_THD_IS_WTSEM(tp)) {
342  tp->u1.semp->cnt++;
343  }
344  else if (NIL_THD_IS_SUSP(tp)) {
345  *tp->u1.trp = NULL;
346  }
347  /*lint -restore*/
348  (void) chSchReadyI(tp, MSG_TIMEOUT);
349  }
350  }
351  /* Lock released in order to give a preemption chance on those
352  architectures supporting IRQ preemption.*/
353  chSysUnlockFromISR();
354  tp++;
355  chSysLockFromISR();
356  } while (tp < &nil.threads[CH_CFG_NUM_THREADS]);
357 #else
358  thread_t *tp = &nil.threads[0];
359  systime_t next = (systime_t)0;
360 
361  chDbgAssert(nil.nexttime == port_timer_get_alarm(), "time mismatch");
362 
363  do {
364  systime_t timeout = tp->timeout;
365 
366  /* Is the thread in a wait state with timeout?.*/
367  if (timeout > (systime_t)0) {
368 
369  chDbgAssert(!NIL_THD_IS_READY(tp), "is ready");
370  chDbgAssert(timeout >= (nil.nexttime - nil.lasttime), "skipped one");
371 
372  /* The volatile field is updated once, here.*/
373  timeout -= nil.nexttime - nil.lasttime;
374  tp->timeout = timeout;
375 
376  if (timeout == (systime_t)0) {
377 #if CH_CFG_USE_SEMAPHORES == TRUE
378  /* Timeout on semaphores requires a special handling because the
379  semaphore counter must be incremented.*/
380  if (NIL_THD_IS_WTSEM(tp)) {
381  tp->u1.semp->cnt++;
382  }
383  else {
384 #endif
385  if (NIL_THD_IS_SUSP(tp)) {
386  *tp->u1.trp = NULL;
387  }
388 #if CH_CFG_USE_SEMAPHORES == TRUE
389  }
390 #endif
391  (void) chSchReadyI(tp, MSG_TIMEOUT);
392  }
393  else {
394  if (timeout <= (systime_t)(next - (systime_t)1)) {
395  next = timeout;
396  }
397  }
398  }
399 
400  /* Lock released in order to give a preemption chance on those
401  architectures supporting IRQ preemption.*/
402  chSysUnlockFromISR();
403  tp++;
404  chSysLockFromISR();
405  } while (tp < &nil.threads[CH_CFG_NUM_THREADS]);
406 
407  nil.lasttime = nil.nexttime;
408  if (next > (systime_t)0) {
409  nil.nexttime += next;
410  port_timer_set_alarm(nil.nexttime);
411  }
412  else {
413  /* No tick event needed.*/
414  port_timer_stop_alarm();
415  }
416 #endif
417 }

498  * @brief Realtime window test.
499  * @details This function verifies if the current realtime counter value
500  * lies within the specified range or not. The test takes care
501  * of the realtime counter wrapping to zero on overflow.
502  * @note When start==end then the function returns always true because the
503  * whole time range is specified.
504  * @note This function is only available if the port layer supports the
505  * option @p PORT_SUPPORTS_RT.
506  *
507  * @param[in] cnt the counter value to be tested
508  * @param[in] start the start of the time window (inclusive)
509  * @param[in] end the end of the time window (non inclusive)
510  * @retval true current time within the specified time window.
511  * @retval false current time not within the specified time window.
512  *
513  * @xclass
514  */
515 bool chSysIsCounterWithinX(rtcnt_t cnt, rtcnt_t start, rtcnt_t end) {
516 
517  return (bool)((cnt - start) < (end - start));
518 }

613 /**
614  * @brief Puts the current thread to sleep into the specified state with
615  * timeout specification.
616  * @details The thread goes into a sleeping state, if it is not awakened
617  * explicitly within the specified system time then it is forcibly
618  * awakened with a @p NIL_MSG_TMO low level message.
619  *
620  * @param[in] newstate the new thread state or a semaphore pointer
621  * @param[in] timeout the number of ticks before the operation timeouts.
622  * the following special values are allowed:
623  * - @a TIME_INFINITE no timeout.
624  * .
625  * @return The wakeup message.
626  * @retval NIL_MSG_TMO if a timeout occurred.
627  *
628  * @sclass
629  */
630 msg_t chSchGoSleepTimeoutS(tstate_t newstate, systime_t timeout) {
631  thread_t *ntp, *otp = nil.current;
632 
633  chDbgCheckClassS();
634 
635  chDbgAssert(otp != &nil.threads[CH_CFG_NUM_THREADS],
636  "idle cannot sleep");
637 
638  /* Storing the wait object for the current thread.*/
639  otp->state = newstate;
640 
641 #if CH_CFG_ST_TIMEDELTA > 0
642  if (timeout != TIME_INFINITE) {
643  systime_t abstime;
644 
645  /* TIMEDELTA makes sure to have enough time to reprogram the timer
646  before the free-running timer counter reaches the selected timeout.*/
647  if (timeout < (systime_t)CH_CFG_ST_TIMEDELTA) {
648  timeout = (systime_t)CH_CFG_ST_TIMEDELTA;
649  }
650 
651  /* Absolute time of the timeout event.*/
652  abstime = chVTGetSystemTimeX() + timeout;
653 
654  if (nil.lasttime == nil.nexttime) {
655  /* Special case, first thread asking for a timeout.*/
656  port_timer_start_alarm(abstime);
657  nil.nexttime = abstime;
658  }
659  else {
660  /* Special case, there are already other threads with a timeout
661  activated, evaluating the order.*/
662  if (chVTIsTimeWithinX(abstime, nil.lasttime, nil.nexttime)) {
663  port_timer_set_alarm(abstime);
664  nil.nexttime = abstime;
665  }
666  }
667 
668  /* Timeout settings.*/
669  otp->timeout = abstime - nil.lasttime;
670  }
671 #else
672 
673  /* Timeout settings.*/
674  otp->timeout = timeout;
675 #endif
676 
677  /* Scanning the whole threads array.*/
678  ntp = nil.threads;
679  while (true) {
680  /* Is this thread ready to execute?*/
681  if (NIL_THD_IS_READY(ntp)) {
682  nil.current = nil.next = ntp;
683  if (ntp == &nil.threads[CH_CFG_NUM_THREADS]) {
684  CH_CFG_IDLE_ENTER_HOOK();
685  }
686  port_switch(ntp, otp);
687  return nil.current->u1.msg;
688  }
689 
690  /* Points to the next thread in lowering priority order.*/
691  ntp++;
692  chDbgAssert(ntp <= &nil.threads[CH_CFG_NUM_THREADS],
693  "pointer out of range");
694  }
695 }
