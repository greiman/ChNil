139 /**
140  * @brief Allocates a block of memory from the heap by using the first-fit
141  * algorithm.
142  * @details The allocated block is guaranteed to be properly aligned for a
143  * pointer data type.
144  *
145  * @param[in] heapp pointer to a heap descriptor or @p NULL in order to
146  * access the default heap.
147  * @param[in] size the size of the block to be allocated. Note that the
148  * allocated block may be a bit bigger than the requested
149  * size for alignment and fragmentation reasons.
150  * @return A pointer to the allocated block.
151  * @retval NULL if the block cannot be allocated.
152  *
153  * @api
154  */
155 static inline void *chHeapAlloc(memory_heap_t *heapp, size_t size) {
156 
157  return chHeapAllocAligned(heapp, size, CH_HEAP_ALIGNMENT);
158 }
