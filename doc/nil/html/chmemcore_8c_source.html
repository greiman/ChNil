20 /**
21  * @file chmemcore.c
22  * @brief Core memory manager code.
23  *
24  * @addtogroup memcore
25  * @details Core Memory Manager related APIs and services.
26  * <h2>Operation mode</h2>
27  * The core memory manager is a simplified allocator that only
28  * allows to allocate memory blocks without the possibility to
29  * free them.<br>
30  * This allocator is meant as a memory blocks provider for the
31  * other allocators such as:
32  * - C-Runtime allocator (through a compiler specific adapter module).
33  * - Heap allocator (see @ref heaps).
34  * - Memory pools allocator (see @ref pools).
35  * .
36  * By having a centralized memory provider the various allocators
37  * can coexist and share the main memory.<br>
38  * This allocator, alone, is also useful for very simple
39  * applications that just require a simple way to get memory
40  * blocks.
41  * @pre In order to use the core memory manager APIs the @p CH_CFG_USE_MEMCORE
42  * option must be enabled in @p chconf.h.
43  * @note Compatible with RT and NIL.
44  * @{
45  */

96 /**
97  * @brief Allocates a memory block.
98  * @details The allocated block is guaranteed to be properly aligned to the
99  * specified alignment.
100  *
101  * @param[in] size the size of the block to be allocated.
102  * @param[in] align desired memory alignment
103  * @return A pointer to the allocated memory block.
104  * @retval NULL allocation failed, core memory exhausted.
105  *
106  * @iclass
107  */
108 void *chCoreAllocAlignedI(size_t size, unsigned align) {
109  uint8_t *p;
110 
111  chDbgCheckClassI();
112  chDbgCheck(MEM_IS_VALID_ALIGNMENT(align));
113 
114  size = MEM_ALIGN_NEXT(size, align);
115  p = (uint8_t *)MEM_ALIGN_NEXT(nextmem, align);
116 
117  if (((size_t)endmem - (size_t)p) < size) {
118  return NULL;
119  }
120  nextmem = p + size;
121 
122  return p;
123 }

125 /**
126  * @brief Allocates a memory block.
127  * @details The allocated block is guaranteed to be properly aligned to the
128  * specified alignment.
129  *
130  * @param[in] size the size of the block to be allocated
131  * @param[in] align desired memory alignment
132  * @return A pointer to the allocated memory block.
133  * @retval NULL allocation failed, core memory exhausted.
134  *
135  * @api
136  */
137 void *chCoreAllocAligned(size_t size, unsigned align) {
138  void *p;
139 
140  chSysLock();
141  p = chCoreAllocAlignedI(size, align);
142  chSysUnlock();
143 
144  return p;
145 }
