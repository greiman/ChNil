73 /**
74  * @brief Data part of a static mailbox initializer.
75  * @details This macro should be used when statically initializing a
76  * mailbox that is part of a bigger structure.
77  *
78  * @param[in] name the name of the mailbox variable
79  * @param[in] buffer pointer to the mailbox buffer area
80  * @param[in] size size of the mailbox buffer area
81  */
82 #define _MAILBOX_DATA(name, buffer, size) { \
83  (msg_t *)(buffer), \
84  (msg_t *)(buffer) + size, \
85  (msg_t *)(buffer), \
86  (msg_t *)(buffer), \
87  _SEMAPHORE_DATA(name.fullsem, 0), \
88  _SEMAPHORE_DATA(name.emptysem, size), \
89 }

146 /**
147  * @brief Returns the number of free message slots into a mailbox.
148  * @note Can be invoked in any system state but if invoked out of a locked
149  * state then the returned value may change after reading.
150  * @note The returned value can be less than zero when there are waiting
151  * threads on the internal semaphore.
152  *
153  * @param[in] mbp the pointer to an initialized mailbox_t object
154  * @return The number of empty message slots.
155  *
156  * @iclass
157  */
158 static inline cnt_t chMBGetFreeCountI(mailbox_t *mbp) {
159 
160  chDbgCheckClassI();
161 
162  return chSemGetCounterI(&mbp->emptysem);
163 }
164 
165 /**
166  * @brief Returns the number of used message slots into a mailbox.
167  * @note Can be invoked in any system state but if invoked out of a locked

169  * @note The returned value can be less than zero when there are waiting
170  * threads on the internal semaphore.
171  *
172  * @param[in] mbp the pointer to an initialized mailbox_t object
173  * @return The number of queued messages.
174  *
175  * @iclass

177 static inline cnt_t chMBGetUsedCountI(mailbox_t *mbp) {
178

184 /**
185  * @brief Returns the next message in the queue without removing it.
186  * @pre A message must be waiting in the queue for this function to work
187  * or it would return garbage. The correct way to use this macro is
188  * to use @p chMBGetFullCountI() and then use this macro, all within
189  * a lock state.
190  *
191  * @param[in] mbp the pointer to an initialized mailbox_t object
192  * @return The next message in queue.
193  *
194  * @iclass
195  */
196 static inline msg_t chMBPeekI(mailbox_t *mbp) {
197 
198  chDbgCheckClassI();
199 
200  return *mbp->rdptr;
201 }
