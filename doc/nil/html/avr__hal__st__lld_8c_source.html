293 #if (OSAL_ST_MODE == OSAL_ST_MODE_FREERUNNING) || defined(__DOXYGEN__)
294 
295  /*
296  * Periodic mode uses Timer 1 (16 bit).
297  */
298 
299  /* CTC mode, no clock source */
300  TCCR1A = 0;
301  TCCR1B = _BV(WGM12);
302 
303  /* start disabled */
304  TCCR1C = 0;
305  OCR1A = 0;
306  TCNT1 = 0;
307  TIFR_REG = _BV(OCF1A); /* Reset pending. */
308  TIMSK_REG = 0;
309  TCCR1B = PRESCALER;
310 
311 #endif /* OSAL_ST_MODE == OSAL_ST_MODE_FREERUNNING */
312 
313 #if (OSAL_ST_MODE == OSAL_ST_MODE_PERIODIC) || defined(__DOXYGEN__)
314 
315  /*
316  * Periodic mode uses Timer 0 (8 bit).
317  */
318 #if defined(TCCR0B) /* Timer has multiple output comparators */
319  TCCR0A = (1 << WGM01) | (0 << WGM00) | /* CTC mode. */
320  (0 << COM0A1) | (0 << COM0A0) | /* OC0A disabled. */
321  (0 << COM0B1) | (0 << COM0B0); /* OC0B disabled. */
322  TCCR0B = (0 << WGM02) | AVR_TIMER_PRESCALER_BITS; /* CTC mode. */
323  OCR0A = AVR_TIMER_COUNTER - 1;
324  TCNT0 = 0; /* Reset counter. */
325  TIFR0 = (1 << OCF0A); /* Reset pending. */
326  TIMSK0 = (1 << OCIE0A); /* IRQ on compare. */
327 
328 #elif defined(TCCR0A) /* AT90CAN doesn't have TCCR0B and slightly different TCCR0A */
329  TCCR0A = (1 << WGM01) | (0 << WGM00) | /* CTC mode. */
330  (0 << COM0A1) | (0 << COM0A0); /* OC0A disabled. */
331  OCR0A = AVR_TIMER_COUNTER - 1;
332  TCNT0 = 0; /* Reset counter. */
333  TIFR0 = (1 << OCF0A); /* Reset pending. */
334  TIMSK0 = (1 << OCIE0A); /* IRQ on compare. */
335 
336 #elif defined(TCCR0) /* Timer has single output comparator */
337  TCCR0 = (1 << WGM01) | (0 << WGM00) | /* CTC mode. */
338  (0 << COM01) | (0 << COM00) | /* OC0A disabled. */
339  AVR_TIMER_PRESCALER_BITS;
340  OCR0 = AVR_TIMER_COUNTER - 1;
341  TCNT0 = 0; /* Reset counter. */
342  TIFR = (1 << OCF0); /* Reset pending. */
343  TIMSK = (1 << OCIE0); /* IRQ on compare. */
344 #else
345  #error "Neither TCCR0A nor TCCR0 registers are defined"
346 #endif
347 
348 #endif /* OSAL_ST_MODE == OSAL_ST_MODE_PERIODIC */
349 
350 }
351
